
FROM pythonfrom arepl_dump import dump:3.11-slim

# Set the working directory in the container
WORKDIR /app/backend

# Copy the dependency files first to leverage Docker caching
COPY requirements.txt .

# Install Python dependencies
RUN pip install --no-cache-dir -r requirements.txt

# Copy the rest of the backend files (including app.py and model artifacts)
# Assuming your model files (e.g., model.joblib, tfidf_vectorizer.pkl) are in this directory
COPY . .

# Expose the port Uvicorn runs on
EXPOSE 8000

# Command to run the application using Uvicorn
# The --host 0.0.0.0 is crucial for Docker to listen on all interfaces
CMD ["uvicorn", "app:app", "--host", "0.0.0.0", "--port", "8000"]
